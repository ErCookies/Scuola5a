Query:

1. *Trova i film con più di 200 milioni di ore viste totali. Intestazione tabella: 'Title', 'total'*:
 	SELECT DISTINCT m.title AS Title, SUM(v.hours\_viewed) AS total FROM view\_summary AS v
 	INNER JOIN movie AS m ON m.id=v.movie\_id
 	GROUP BY m.id HAVING total > 200000000;

2. *Quanti film hanno più di 200 milioni di ore viste totali*:
 	SELECT COUNT(\*) FROM (SELECT \* FROM view\_summary AS v
 	INNER JOIN movie AS m ON m.id=v.movie\_id
 	GROUP BY m.id HAVING SUM(v.hours\_viewed) > 200000000);

3. *Elenca le serie TV con almeno 20 stagioni*:
 	SELECT t.title FROM tv\_show as t
 	INNER JOIN season AS s ON s.tv\_show\_id=t.id
 	GROUP BY t.id, t.title HAVING COUNT(\*) > 20;
